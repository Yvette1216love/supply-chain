    const cls = d.alert.type === 'danger' ? 'd-alert--danger' : d.alert.type === 'warn' ? 'd-alert--warn' : 'd-alert--ok';
    const icon = d.alert.type === 'danger' ? '⚠' : d.alert.type === 'warn' ? '⚡' : '✓';
    html += `<div class="d-alert ${cls}"><span class="d-alert-icon">${icon}</span><span>${d.alert.text}</span></div>`;
  }

  // Supply chain
  if (d.chain && d.chain.length) {
    html += `<div class="d-section"><div class="d-section-title">供應鏈流向</div><div class="d-chain">`;
    d.chain.forEach((s, i) => {
      html += `<span class="d-chain-step" style="background:${s.c}18;border-color:${s.c}44;color:${s.c}">${s.n}</span>`;
      if (i < d.chain.length - 1) html += `<span class="d-chain-arrow">→</span>`;
    });
    html += `</div></div>`;
  }

  // Suppliers
  if (d.suppliers && d.suppliers.length) {
    html += `<div class="d-section"><div class="d-section-title">主要廠商 / 供應商</div><div class="d-suppliers">`;
    d.suppliers.forEach(s => {
      html += `<div class="d-sup">
        <div class="d-sup-flag">${s.flag}</div>
        <div class="d-sup-body">
          <div class="d-sup-name">${s.name}${s.share ? `<span class="d-sup-share">${s.share}</span>` : ''}</div>
          <div class="d-sup-note">${s.note}</div>
        </div>
      </div>`;
    });
    html += `</div></div>`;
  }

  // Specs
  if (d.specs && d.specs.length) {
    html += `<div class="d-section"><div class="d-section-title">技術規格 / 關鍵數據</div>
    <table class="d-table"><thead><tr><th>項目</th><th>數值 / 說明</th></tr></thead><tbody>`;
    d.specs.forEach(s => {
      html += `<tr><td><span class="d-mat-name">${s.l}</span></td><td>${s.v}</td></tr>`;
    });
    html += `</tbody></table></div>`;
  }

  document.getElementById('d-body').innerHTML = html;

  // Open
  document.getElementById('drawer').classList.add('open');
  document.getElementById('canvas').classList.add('drawer-open');
  document.getElementById('overlay').classList.add('show');
}

function closeDrawer() {
  document.getElementById('drawer').classList.remove('open');
  document.getElementById('canvas').classList.remove('drawer-open');
  document.getElementById('overlay').classList.remove('show');
  document.querySelectorAll('.node-card').forEach(c => c.classList.remove('active'));
  currentNode = null;
}
